fleet_event = {
	id = pki_paragons.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = {
			exists = planet
			planet_owner  = { has_origin = origin_katzen}
			any_fleet_in_orbit = {
				is_ship_class = shipclass_science_ship
				owner = { has_origin = origin_katzen }
				leader = {
					has_trait = leader_trait_schulz
				}
			}
		}
	}

	immediate = {
		from = {
			add_modifier = {
				modifier = meowrine_world
				days = -1
			}
			planet_event = {
				id = pki_paragons.3
				days = 30
			}
		}
	}
}

planet_event = {
	id = pki_paragons.3
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				any_fleet_in_orbit = {
					is_ship_class = shipclass_science_ship
					owner = { has_origin = origin_katzen }
					leader = {
						has_trait = leader_trait_schulz
					}
				}
			}
			add_modifier = {
				modifier = meowrine_world
				days = -1
			}
			planet_event = {
				id = pki_paragons.3
				days = 30
			}
		}
		else = {
			remove_modifier = meowrine_world
		}
	}

}
